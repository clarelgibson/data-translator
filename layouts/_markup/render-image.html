{{/* Image render hook. Turns ![alt](file.png "caption") into an optimised, captioned figure. .Text -> alt text .Destination -> image filename (a page-bundle resource) .Title -> caption Non-bundle images (e.g. external URLs) fall back to a plain . */}} {{- $img := .PageInner.Resources.GetMatch .Destination -}} {{- if $img -}} {{- $sizes := slice 480 720 1080 1440 -}} {{- $srcset := slice -}} {{- range $sizes -}} {{- if lt . $img.Width -}} {{- $r := $img.Resize (printf "%dx Lanczos" .) -}} {{- $srcset = $srcset | append (printf "%s %dw" $r.RelPermalink .) -}} {{- end -}} {{- end -}} {{- $srcset = $srcset | append (printf "%s %dw" $img.RelPermalink $img.Width) -}} {{- $display := $img -}} {{- if ge $img.Width 720 -}}{{- $display = $img.Resize "720x Lanczos" -}}{{- end -}} {{- else -}} {{- end -}}
